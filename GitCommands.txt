git --version
git init (master branch by default) or git init -b main
git status 
------
git add "file1"
git commit -m "msg or title"
------(you can skip "git add" by below cmd)
git commit -a -m "4th commit"

git log (list of all commits)
git diff (to see what are changes compared to previous files)

git rm --cached filename (remove file)

git clone RepoURL(https/ssh)  (clone all the project file in remote repo to local repo)

git branch -M main  (rename existing branch(master) to main)

git remote add origin git@github.com:sahas0212/Git-Commands.git (adding nickname "origin" for repo path git@g...)
git push -u origin main (tell github to push main branch into origin(repo path))
------------(after the above two commands u can directly write "git push" and "git pull")

git push (push to remote repo)

git remote -v (to see which url is given to origin(or any given name))

git tag -a tagname -m "message" (or) git tag tagname -m "message" (to create tags)
git tag  (list of tags)
git show tagname  (given details about that tag)
git push origin tagname  (to push tags to GitHub)

git checkout -b newbranchname  / git switch -c newbranchname (create branch) 
git checkout newbranch / git switch newbranch (Switch branch)
git switch -  (switch to previously used branch)
git branch (see all branches present)
git branch -d branchname (delete branch)

git pull origin main  (get code from main)
git merge branchname  (merge codes in branch to main)
git rebase branchname  (to get linear history)

git log --graph (to visualize branches)
